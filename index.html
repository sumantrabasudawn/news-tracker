        // 1. TICKER ENGINE: Fetches market data [cite: 48, 49]

        async function loadTicker() {

            try {

                // Cache-busting added to ensure real-time updates

                const res = await fetch(`output/market_data.csv?t=${Date.now()}`);

                if (!res.ok) throw new Error('Network response was not ok');

                

                const text = await res.text();

                const rows = text.trim().split("\n").slice(1);

                let tickerHTML = "";


                rows.forEach(row => {

                    if (!row) return;

                    const [asset, price, change] = row.split(",");

                    

                    const changeNum = parseFloat(change) * 100;

                    const direction = changeNum >= 0 ? "up" : "down";

                    const arrow = changeNum >= 0 ? "▲" : "▼";


                    tickerHTML += `

                        <span class="${direction}">

                            ${asset} ${parseFloat(price).toLocaleString(undefined, {minimumFractionDigits: 2})} 

                            ${arrow} ${Math.abs(changeNum).toFixed(2)}%

                        </span>

                    `;

                });


                // Double the content for a seamless infinite loop

                document.getElementById("ticker-track").innerHTML = tickerHTML + tickerHTML;


            } catch (err) {

                console.error("Ticker error:", err);

                document.getElementById("ticker-track").innerHTML = "<span>Market Data Temporarily Unavailable</span>";

            }

        }


        // 2. ROUTING ENGINE: Redirects to latest report [cite: 53]

        // Currently hardcoded to April 26 as per system state [cite: 35]

        const latestReport = "reports/morning_intelligence_dashboard_26_april_2026.html";


        function goToLatest() {

            window.location.href = latestReport;

        }


        // Initialize System

        loadTicker();

        setInterval(loadTicker, 60000); // Auto-refresh every 60s


        // Auto-redirect after 3 seconds

        setTimeout(() => {

            document.getElementById("status-msg").innerText = "Redirecting to Intelligence Dashboard...";

            goToLatest();

        }, 3000);

    </script>
