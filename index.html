  <script>

    async function loadTicker() {

      try {

        const res = await fetch("output/market_data.csv");

        const text = await res.text();


        const rows = text.split("\n").slice(1);

        let tickerHTML = "";


        rows.forEach(row => {

          if (!row) return;


          const [asset, price, change] = row.split(",");


          const changeNum = parseFloat(change);

          const direction = changeNum >= 0 ? "up" : "down";

          const arrow = changeNum >= 0 ? "▲" : "▼";


          tickerHTML += `

            <span class="${direction}">

              ${asset} ${arrow} ${changeNum.toFixed(2)}%

            </span>

          `;

        });


        // Duplicate content for smooth infinite scroll

        document.getElementById("ticker-track").innerHTML = tickerHTML + tickerHTML;


      } catch (err) {

        console.error("Ticker error:", err);

      }

    }


    loadTicker();

    setInterval(loadTicker, 60000);

  </script>
